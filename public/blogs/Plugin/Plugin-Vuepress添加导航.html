          </a>
        </div>
      </div>
    </div>
  </div>
</template>

# Step3 可扩展性

如果你想添加导航，可以在navList中添加,格式如下

    navList: [ // 整体作为一个数组,用于循环渲染
        { // 每个数组下，有多个分类
          title: "动漫 | 漫画 | 兴趣", // 当前分类的标题
          arr: [ // 当前分类下的导航
            {
              nav: "bilibili", // 导航名
              nav_url: "https://www.bilibili.com/", // 导航链接
              id: "1", // 导航id
            },
          ],
        },
    ]
